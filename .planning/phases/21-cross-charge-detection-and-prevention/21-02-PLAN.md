    ```python
    def _apply_cross_charge_guard(
        self, h_snap: ControllerSnapshot, v_snap: ControllerSnapshot,
        h_cmd: ControllerCommand, v_cmd: ControllerCommand,
    ) -> tuple[ControllerCommand, ControllerCommand]:
        """Check for cross-charge and mitigate if detected."""
        if self._cross_charge_detector is None:
            return h_cmd, v_cmd
        xc_state = self._cross_charge_detector.check(h_snap, v_snap)
        if xc_state.detected:
            h_cmd, v_cmd = self._cross_charge_detector.mitigate(xc_state, h_cmd, v_cmd)
            # Log decision entry
            entry = DecisionEntry(
                timestamp=datetime.now(tz=timezone.utc).isoformat(),
                trigger="cross_charge_prevention",
                huawei_role=h_cmd.role.value,
                victron_role=v_cmd.role.value,
                p_target_w=0.0,
                huawei_allocation_w=h_cmd.target_watts,
                victron_allocation_w=v_cmd.target_watts,
                pool_status=self._state.pool_status if self._state else "NORMAL",
                reasoning=f"Cross-charge detected: {xc_state.source_system} discharging "
                          f"({xc_state.source_power_w:.0f}W) into {xc_state.sink_system} "
                          f"({xc_state.sink_power_w:.0f}W), grid={xc_state.net_grid_power_w:.0f}W. "
                          f"Forced {xc_state.sink_system} to HOLDING.",
            )
            self._decisions.append(entry)
            logger.warning(
                "cross-charge detected: %s->%s, waste=%.0fW, grid=%.0fW — forced %s to HOLDING",
                xc_state.source_system, xc_state.sink_system,
                min(xc_state.source_power_w, xc_state.sink_power_w),
                xc_state.net_grid_power_w, xc_state.sink_system,
            )
            # Telegram alert — "first per episode only" is enforced by
            # TelegramNotifier's built-in 300s per-category cooldown (see
            # notifier.py send_alert). ALERT_CROSS_CHARGE is a distinct
            # category so the cooldown window naturally aligns with the
            # episode_reset_s=300s in CrossChargeDetector. No additional
            # episode-transition gate is needed here because:
            #   1. First alert fires immediately (no prior cooldown).
            #   2. Subsequent calls within the same episode are suppressed
            #      by the 300s cooldown.
            #   3. After 300s clear (episode resets), the cooldown also
            #      expires, allowing the next episode's first alert through.
            if self._notifier is not None:
                try:
                    self._notifier.send_alert(
                        ALERT_CROSS_CHARGE,
                        f"Cross-charge detected: {xc_state.source_system} discharging "
                        f"into {xc_state.sink_system}. Forced {xc_state.sink_system} to HOLDING.",
                    )
                except Exception as exc:
                    logger.warning("cross-charge telegram alert failed: %s", exc)
        return h_cmd, v_cmd
    ```
